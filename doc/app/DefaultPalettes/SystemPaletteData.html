def palettes
  palettes = []
  (1..4).each do |index|
    data = load_file("lib/tasks/data/sample_palettes#{index}.json")
    palette = SimplePalette.new(data["name"])
    actions = data['actions']
    buttons = []
    actions.each do |action|
      buttons << make_button(action)
    end
    palette.buttons = buttons
    palettes << palette
  end
  palettes
end
